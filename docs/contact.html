Integration

Add Chatwoot widget to your deployment:

<script>
window.chatwootSettings = {
  position: "bottom-right",
  language: "en",
  type: "standard"
};
(function(){
  var d=document, w=window, t=w.Chatwoot;
  if(typeof t!="function"){
    var s=document.createElement("script");
    s.src="https://cdn.chatwoot.com/widgets.js";
    s.async=true;
    document.head.appendChild(s);
  }
})();
</script>
