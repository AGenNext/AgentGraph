Create Enterprise Agent

from orchestrator import create_agent

# Project Driver - uses gpt-4o by default
driver = create_agent("proj-driver", "project_driver", 
    config={"projects": ["proj-a", "proj-b"]})

# Override orchestrator/LLM in config
lead = create_agent("team-lead", "team_lead",
    config={"manages": ["a", "b"], "orchestrator": "langgraph", "llm": "claude-3.5-sonnet"})
